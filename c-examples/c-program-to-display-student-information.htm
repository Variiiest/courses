C Program to Sort strings Lexicographically (Dictionary Order)

#include <stdio.h>
struct student
{
    char name[50];
    int roll;
} s[10];
int main()
{
    int i;
    printf("Enter information of students:\n");
    // storing information
    for(i=0; i<10; ++i)
    {
        s[i].roll = i+1;
        printf("\nFor roll number%d,\n",s[i].roll);
        printf("Enter name: ");
        scanf("%s",s[i].name);
        printf("\n");
    }
    printf("Information:\n\n");
    for(i=0; i<10; ++i)
    {
        printf("\nRoll number: %d\n",i+1);
        printf("Name: ");
        puts(s[i].name);
        printf("\n");
    }
    return 0;
}
