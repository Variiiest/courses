Recursion

A recursive function is defined as a function on that calles itself to solve a problem

void recurse()
{
    ... .. ...
    recurse();
    ... .. ...
}

int main()
{
    ... .. ...
    recurse();
    ... .. ...
}

Example :Sum of n natural number using recursion 

#include <stdio.h>
int sum(int n);
int main()
{
    int number, result;
    printf("Enter a positive integer: ");
    scanf("%d", &number);
    result = sum(number);
    printf("sum = %d", result);
    return 0;
}
int sum(int num)
{
    if (num!=0)
        return num + sum(num-1); // sum() function calls itself
    else
        return num;
}

Cases

Base case :

To solve a problem without further calls to some function.

Recursive case :

To solve from division of problem and create function to call itself.

Factorial program using recursion

Define:

Base case ------- 1!=1

Recursive case ------- n!=n*factorial(n-1)

#include <stdio.h>
int main()
{
    int n, i;
    unsigned long long factorial = 1   // Base case;
    printf("Enter an integer: ");
    scanf("%d",&n);
    // show error if the user enters a negative integer
    if (n < 0)
        printf("Error! Factorial of a negative number doesn't exist.");
    else
    {
        for(i=1; i<=n; ++i)
        {
            factorial *= i;              // factorial = factorial*i Recursive case;
        }
        printf("Factorial of %d = %llu", n, factorial);
    }
    return 0;
}

Direct recursion

Called directly

#include <stdio.h>
 int main()
 {
   int n, i = 0, term;
   printf("Enter the number of terms\n");
   scanf("%d",&n);
   printf("fibonacci series\n");
   for(i = 0; i < n; i++ )
   {
     term = fibonacci(i);
     printf("%d\t",term);
   } 
   return 0;
 }
 int fibonacci(int n)
 {
   if ( n == 0 )
     return 0;
   else if ( n == 1 )
     return 1;
   else  return ( fibonacci(n–1) + fibonacci(n–2) );
 }

{

 if(n==1)
