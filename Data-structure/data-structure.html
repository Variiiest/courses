Data-Structure

A storage of data elements under a name called as DS.

char vowel[5]={'a','e','i','o','u'};

vowel is the name of data structure.And stored vowels are data elements.

Data-Structure Terms

  • Data: A particular value or set of values.
  • Record: Record is collection of data.
  • File: File is collection of records.

Searching

#include<stdio.h>
int main()
{
   int array[100], search, c, n, count = 0;
   
   printf("Enter number of elements in array\n");
   scanf("%d", &n);
   
   printf("Enter %d numbers\n", n);
   
   for (c = 0; c < n; c++)
      scanf("%d", &array[c]);
     
   printf("Enter a number for searching\n");
   scanf("%d", &search);
   
   for (c = 0; c < n; c++) {
      if (array[c] == search) {
         printf("%d is present at location %d.\n", search, c+1);
         count++;
      }
   }
   if (count == 0)
      printf("%d isn't present in the array.\n", search);
   else
      printf("%d is present %d times in the array.\n", search, count);
     
   return 0; 
}

   //Delete a node from given linked list
   struct Node 
   { 
	int data; 
	struct Node *next; 
   };
   void push(struct Node** head, int new_data) 
   { 
	struct Node* new_node = (struct Node*) malloc(sizeof(struct Node)); 
	new_node->data = new_data; 
	new_node->next = (*head); 
	(*head) = new_node; 
   } 

   void deleteNode(struct Node **head, int key) 
   { 
	struct Node* temp = *head, *prev; 
	if (temp != NULL && temp->data == key) 
	{ 
		*head = temp->next;
		free(temp);	
		return; 
	} 
	while (temp != NULL && temp->data != key) 
	{ 
		prev = temp; 
		temp = temp->next; 
	} 

	if (temp == NULL) return;
	prev->next = temp->next; 
	free(temp);
  } 
  void printList(struct Node *node) 
  { 
	while (node != NULL) 
	{ 
		printf(" %d ", node->data); 
		node = node->next; 
	} 
  }

Merging

#include<stdio.h>
#include<conio.h>
void main()
{
	int arr1[50], arr2[50], size1, size2, size, i, j, k, merge[100];
	printf("Enter Array 1 Size : ");
	scanf("%d",&size1);
	printf("Enter Array 1 Elements : ");
	for(i=0; i<size1; i++)
	{
		scanf("%d",&arr1[i]);
	}
	printf("Enter Array 2 Size : ");
	scanf("%d",&size2);
	printf("Enter Array 2 Elements : ");
	for(i=0; i<size2; i++)
	{
		scanf("%d",&arr2[i]);
	}
	for(i=0; i<size1; i++)
	{
		merge[i]=arr1[i];
	}
	size=size1+size2;
	for(i=0, k=size1; k<size && i<size2; i++, k++)
	{
		merge[k]=arr2[i];
	}
	printf("Now the new array after merging is :\n");
	for(i=0; i<size; i++)
	{
		printf("%d  ",merge[i]);
	}
}

ABSTRACT DATA TYPE

ABSTRACT DATA TYPE

Abstract data type is a look of objects whose logical behaviour is defined by a set of values and set of operations.

ADVANTAGE

  • Localization of change
  • Flexibility

OPERATION

Some operations that are often spechified for Abstract data type.
  • compare()
  • hash()
  • print() or show()
