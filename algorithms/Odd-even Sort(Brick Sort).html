func_tion oddEvenSort(array)
 {   
   func_tion  swap( array, i, j )
   {    
      var temp = array[i];   
      array[i] = array[j]; 
     array[j] = temp;  
  } 
 
  var sorted = false;
   while(!sorted)  
 {     
    sorted = true;    
    for(var i = 1; i < array.size-1; i += 2)    
 {      
    if(array[i] > array[i+1])      
    {        
      swap(array, i, i+1);         
      sorted = false;      
    }    
 } 
 for(var i = 0; i < array.size-1; i += 2)  
   {

     {        
     swap(array, i, i+1);        
     sorted = false;   
    }   
  }  
 }
 }
