Array Multiplication Algo

Step 1: Input ARR1,ARR2 
Step 2: SET SUM =0,ARR3
Step :  IF ARR1 COLUMN IS NOT EQUAL TO ARR2 ROW
        PRINT MULTIPLICATION IMPOSSIBLE
Step 3: SUM=SUM+ARR1 ELEMENTS *ARR2 ELEMENTS
Step 4: ARR3=SUM
Step 5: PRINT ARR3

Algo Implemation in C

        
#include <stdio.h>
int main()
{
  int r1, c1, r2, c2, i, j, k, sum = 0;
  int matrix1[10][10], matrix2[10][10], matrix3[10][10];
 
  printf("Enter number of rows and columns of matrix1 \n");
  scanf("%d%d", &r1, &c1);
  printf("Enter elements of matrix1 matrix\n");
 
  for (i = 0; i < r1; i++)
    for (j = 0; j < c1; j++)
      scanf("%d", &matrix1[i][j]);
 
  printf("Enter number of rows and columns of matrix2 \n");
  scanf("%d%d", &r2, &c2);
 
  if (c1 != r2)
    printf("The matrices can't be multiplied \n");
  else
  {
    printf("Enter elements of matrix2\n");
 
    for (i = 0; i < r2; i++)
      for (j = 0; j < c2; j++)
        scanf("%d", &matrix2[i][j]);
 
    for ( i= 0; i < r1; i++) {
      for (j = 0; j < c2; j++) {
        for (k = 0; k < r1; k++) {
          sum = sum + matrix1[i][k]*matrix2[k][j];
        }
 
        matrix3[i][j] = sum;
        sum = 0;
      }
    }
 
    printf("Product of the matrices:\n");
    for ( i= 0; i < r1; i++) {
      for (j = 0; j < c2; j++) 
        printf("%d\t", matrix3[i][j]);
 
      printf("\n");
    }
  }
 
  return 0;
}
