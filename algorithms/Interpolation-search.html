Interpolation Search

1:SET LOW = lower_bound,  
    HIGH = upper_bound, POS = –1 
2:    Repeat Steps 3 to 4 while LOW <= HIGH 
3:         SET MID = LOW + (HIGH – LOW) ×  ((VAL – A[LOW]) / (A[HIGH] – A[LOW])) 
4:          IF VAL = A[MID]  
               POS = MID   
               PRINT POS 
               Go to Step 6
            ELSE IF VAL < A[MID]
               SET HIGH = MID – 1
            ELSE
                    SET LOW = MID + 1
            [END OF IF]
       [END OF LOOP] 
5: IF POS = –1       
        PRINT "VALUE IS NOT PRESENT IN THE ARRAY"         
  [END OF IF] 
6: EXIT

Implemation in C

#include<stdio.h> 
int interpolationSearch(int arr[], int n, int val) 
{  
	int l = 0, b = (n - 1); 
	while (l <= b && val >= arr[l] && val <= arr[b]) 
	{ 
		if (l == b){ 
			if (arr[l] == val) return l; 
			return -1; 
		} 
		int pos = l + (((double)(b-l) / (arr[b]-arr[l]))*(val -arr[l])); 
		if (arr[pos] == val) 
			return pos; 
		if (arr[pos] < val) 
			l = pos + 1; 
		else
			b = pos - 1; 
	} 
	return -1; 
} 
int main() 
{ 
	 int arr[10] ,n,i; 
    printf("Enter the size of array");
    scanf("%d",&n);
    printf("\n Enter the Numbers of Array");
    for(i=0;i<n;i++)
    {
    	scanf("%d",&arr[i]);
	}
	int val;
	printf("Enter number to search");
	scanf("%d",&val);
	int result = interpolationSearch(arr, n, val); 
 
	if (result != -1) 
		printf("Number found at position %d", result+1); 
	else
		printf("Number not found."); 
	return 0; 
}
