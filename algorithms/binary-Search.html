Binary Search

Binary search is a search algorithm that finds the position of a given value within a sorted array.

Algo

FUNC_BIN(A, l,b,VAL)
 1: SET BEG=l
      END=b, POS=0 
 2: Repeat 3 and 4 while BEG <= END
 3:      SET MID=(BEG+END)/2 
 4:      IF A[MID]=VAL 
             SET POS=MID+1 
             PRINT POS 
             Go to  6 
         ELSE IF A[MID]>VAL 
              SET END=MID-1 
         ELSE SET BEG=MID+1
             [END OF IF] 
    [END OF LOOP] 
 5: IF POS=0
       PRINT “VALUE DOES NOT EXIST” 
   [END OF IF] 
 6: EXIT

Implemation in C

#include <stdio.h> 
int binarySearch(int arr[], int l, int b, int x) 
{ 
	if (b >= l) { 
		int mid = l + (b - l) / 2; 
		if (arr[mid] == x) 
			return mid+1;  
		if (arr[mid] > x) 
			return binarySearch(arr, l, mid - 1, x); 
		return binarySearch(arr, mid + 1, b, x); 
	} 
	return 0; 
}
