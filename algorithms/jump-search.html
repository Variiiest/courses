Jump Search

Jump search is very useful searching algorithm when the array is sorted.

Implemation in C

#include<stdio.h>
#include<math.h>
int min(int a,int b)
{
	if(a<b)
	 return a;
	else
	  return b;
}
int jump_search(int arr[], int val, int n) 
{ 
	int s = sqrt(n); 
	int p = -1; 
	while (arr[min(s, n)-1] < val) 
	{ 
		p = s; 
		s += sqrt(n); 
		if (p >= n) 
			return -1; 
	} 

	while (arr[p] < val) 
	{ 
		p++; 
		if (p == min(s, n)) 
			return -1; 
	} 
	if (arr[p] == val) 
		return p; 

	return p; 
}
