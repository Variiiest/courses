Shell sort

Shell sort, invented by Donald Shell (1959)is a sorting algorithm that is a generalization of insertion sort.

Algo

 1: SET FLAG=1, GAP_SIZE=N 
 2: Repeat 3 to 6 while FLAG=1 OR GAP_SIZE>1
 3:     SET FLAG = 0
 4:     SET GAP_SIZE=(GAP_SIZE + 1) / 2 
 5:     Repeat 6 for I= to I < (N -GAP_SIZE) 
 6:            IF Arr[I+GAP_SIZE]>Arr[I] 
                   SWAP Arr[I+GAP_SIZE], Arr[I] 
                   SET FLAG = 0
 7: END

Implemation in C

#include <stdio.h>
#include <stdlib.h>
#include <string.h>
void shell_sort(int arr[], int n) {
  int gap = n/3;
  int p = 1;
  int i;
  while(gap > 0) {
    for( i = gap; i < n; i++) {
      int temp = arr[i], j;

      for(j = i; j >= gap && arr[j - gap] > temp; j -= gap) {
        arr[j] = arr[j - gap];
      }

      arr[j] = temp;
    }
    p++;
    gap /= 2;
  }
}
int main() {
  int arr[] = {10, 7, 3, 1, 9, 7, 4, 3};
  int n = sizeof(arr) / sizeof(int),i;
  shell_sort(arr, n);
  for(i=0;i<n;i++)
  printf("%d ",arr[i]);
  return 0;
}
